Updated: 2021-12-25 22:02:18+01:00

in 2013

Determine what is the local PERL5LIB configuration:

LIB=$(
  for d in `tr : ' ' <<<$PERL5LIB
  do
    if [ -w $d ] ; then
      echo $d
    break
  fi
  done)
PREFIX=`dirname $LIB`

Install sequence

#PREFIX=$HOME/cpan
#LIB=$HOME/cpan/lib
tar zxvf $perl-mod-tar
cd $unpacked-src-dir
perl Makefile.PL PREFIX=$PREFIX LIB=$LIB "$@"
make
make test
make install
